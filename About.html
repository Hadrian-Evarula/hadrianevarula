                // Function to handle deletion confirmation using SweetAlert
                function confirmDelete(id) {
                    Swal.fire({
                        title: 'Are you sure?',
                        text: 'You will not be able to recover this user!',
                        icon: 'warning',
                        showCancelButton: true,
                        confirmButtonText: 'Yes, delete it!',
                        cancelButtonText: 'No, keep it'
                    }).then((result) => {
                        if (result.isConfirmed) {
                        // If user confirms deletion, redirect to delete.php with the user's id
                        window.location.href = 'delete.php?id=' + id;
                        }
                    });
                }

Update


        <?php
        // Enable error reporting
        error_reporting(E_ALL);
        ini_set('display_errors', 1);

        include './connectionDB/connection.php';

        // Check if the id is set in the URL
        if (isset($_GET['id'])) {
            $id = $_GET['id'];

            // Fetch user data from the database
            $sql = "SELECT * FROM users WHERE id=$id";
            $result = $conn->query($sql);

            if ($result) {
                if ($result->num_rows == 1) {
                    $row = $result->fetch_assoc();
                    $firstname = $row['firstname']; 
                    $lastname = $row['lastname'];
                    $age = $row['age'];
                } else {
                    echo "No user found with ID: $id";
                    exit();
                }
            } else {
                echo "Error fetching user data: " . $conn->error;
                exit();
            }
        } else {
            echo "ID parameter is missing in the URL";
            exit();
        }

        if ($_SERVER["REQUEST_METHOD"] == "POST" && isset($_POST['id'])) {
            $id = $_POST['id'];
            $newFirstname = $_POST['newFirstname'];
            $newLastname = $_POST['newLastname'];
            $newAge = $_POST['newAge'];

            // Update the record in the database
            $sql = "UPDATE users SET firstname='$newFirstname',lastname='$newLastname',age='$newAge' WHERE id=$id";

            if ($conn->query($sql) === TRUE) {
                header("Location: ./index.php");
                exit();
            } else {
                echo "Error updating record: " . $conn->error;
            }
        }

        // Debugging: Output POST data
        echo "POST Data: <pre>";
        var_dump($_POST);
        echo "</pre>";
        ?>
